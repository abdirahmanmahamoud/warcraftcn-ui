import { Button } from "@/components/ui/warcraftcn/button";
import {
  Dialog,
  DialogClose,
  DialogContent,
  DialogDescription,
  DialogFooter,
  DialogHeader,
  DialogTitle,
  DialogTrigger,
} from "@/components/ui/warcraftcn/dialog";

## Standard

<Dialog>
  <DialogTrigger asChild>
    <Button>Open</Button>
  </DialogTrigger>
  <DialogContent>
    <DialogHeader>
      <DialogTitle>Are you absolutely sure?</DialogTitle>
      <DialogDescription>
        This action cannot be undone. This will permanently delete your account
        and remove your data from our servers.
      </DialogDescription>
    </DialogHeader>
    <DialogFooter>
      <DialogClose asChild>
        <Button className="brightness-75 saturate-50">Maybe Later</Button>
      </DialogClose>
      <Button>I Accept</Button>
    </DialogFooter>
  </DialogContent>
</Dialog>

## Installation

```bash
pnpm dlx shadcn@latest add https://warcraftcn.com/r/dialog.json
```

## Usage

```tsx
import { Dialog } from "@/components/ui/warcraftcn/dialog";

<Dialog>
  <DialogTrigger>Open</DialogTrigger>
  <DialogContent>
    <DialogHeader>
      <DialogTitle>Are you absolutely sure?</DialogTitle>
      <DialogDescription>
        This action cannot be undone. This will permanently delete your account
        and remove your data from our servers.
      </DialogDescription>
    </DialogHeader>
    <DialogFooter>
      <DialogClose asChild>
        <Button className="brightness-75 saturate-50">Cancel</Button>
      </DialogClose>
      <DialogClose asChild>
        <Button>Continue</Button>
      </DialogClose>
    </DialogFooter>
  </DialogContent>
</Dialog>;
```
